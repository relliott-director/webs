  <script>

    let timerInterval;

    let elapsed = 0;

    let running = false;

    let lastStart = null;


    function updateTimerDisplay() {

      let total = elapsed;

      if (running && lastStart) {

        // Get current time in Dublin

        const now = new Date();

        const nowDublin = new Date(now.toLocaleString("en-US", { timeZone: "Europe/Dublin" }));

        total += Math.floor((nowDublin - lastStart) / 1000);

      }

      const hours = String(Math.floor(total / 3600)).padStart(2, '0');

      const minutes = String(Math.floor((total % 3600) / 60)).padStart(2, '0');

      const seconds = String(total % 60).padStart(2, '0');

      document.getElementById('timer').textContent = `${hours}:${minutes}:${seconds}`;

    }


    function startTimer() {

      if (!running) {

        running = true;

        // Use Dublin time for start

        const now = new Date();

        lastStart = new Date(now.toLocaleString("en-US", { timeZone: "Europe/Dublin" }));

        timerInterval = setInterval(updateTimerDisplay, 1000);

      }

    }


    function pauseTimer() {

      if (running) {

        running = false;

        // Use Dublin time for pause

        const now = new Date();

        const nowDublin = new Date(now.toLocaleString("en-US", { timeZone: "Europe/Dublin" }));

        elapsed += Math.floor((nowDublin - lastStart) / 1000);

        clearInterval(timerInterval);

        updateTimerDisplay();

      }

    }


    function resetTimer() {

      running = false;

      elapsed = 0;

      lastStart = null;

      clearInterval(timerInterval);

      updateTimerDisplay();

    }


    // Initialize display

    updateTimerDisplay();

  </script>
